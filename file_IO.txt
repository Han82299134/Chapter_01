import glob
import json
import os

import cv2
from ultralytics import YOLO

folder_path= "../img"
img_path = r'img\1_发动机缸盖_model_pic.jpg'
YOLO_model_path = r'../YOLO_model/1_发动机缸盖_component_model.pt'
component_box_path = r'../YOLO_model/1_发动机缸盖_component_box.json'

def read_image():
    # 第一步：获取所有图片路径
    image_paths = get_all_image_paths(folder_path)
    valid_images = []  # 存储成功读取的图像
    failed_paths = []  # 存储读取失败的路径

    # 第二步：遍历路径，逐个读取图片
    for img_path in image_paths:
        # cv2.imread读取图片（返回None表示读取失败）
        img = cv2.imread(img_path)
        if img is not None:
            # 读取成功，加入有效列表
            valid_images.append(img)
        else:
            # 读取失败，记录路径并提示
            failed_paths.append(img_path)
            print(f"警告：无法读取图片 {img_path}（文件损坏/不是有效图片）")

    return valid_images

def read_yolo_model():

def read_data_from_json(file_path):
    with open(f'{file_path}', 'r', encoding='utf-8') as f:
        datas = json.load(f)
    return datas

# 获取界面UI元素图片的地址路径
def get_icons_file_path(pic_name):
    icons_file_path = f"icon/{pic_name}"

    # 判断模型文件是否存在并加载模型
    if os.path.exists(icons_file_path) and os.path.isfile(icons_file_path):
        return icons_file_path
    else:
        print(f"图标 {icons_file_path} 不存在。")

def get_all_image_paths(folder_path="img"):
    """
    获取指定文件夹中所有图片的绝对路径列表
    """
    # 定义常见的图片扩展名
    image_extensions = ["*.jpg", "*.jpeg", "*.png", "*.bmp", "*.gif", "*.tiff"]
    image_paths = []

    for ext in image_extensions:
        pattern = os.path.join(folder_path, ext)
        matched_paths = glob.glob(pattern)
        image_paths.extend(matched_paths)

    # 转换为绝对路径
    image_paths = [os.path.abspath(path) for path in image_paths]
    return image_paths
